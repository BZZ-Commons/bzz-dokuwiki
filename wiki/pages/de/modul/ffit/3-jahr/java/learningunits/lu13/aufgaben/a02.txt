==== LU13.A02 Ajax Page Collector ====

Machen Sie einen Cherry-Pick des Commits ''a05c359'', um den fertigen ''WikiPageCollectorService'' zu erhalten.

Nutzen Sie den folgenden Code, um die Funktionalität zu überprüfen und mit der Hierarchie der Übersichtsseite zu vergleichen.

https://wiki.bzz.ch/start?idx=de%3Amodul%3Affit%3A3-jahr%3Ajava%3Alearningunits%3Alu11%3A

<code java>
package ch.bzz.rag;

import ch.bzz.rag.service.*;
import lombok.extern.slf4j.Slf4j;

import java.util.Set;

@Slf4j
public class ApplicationMain {

    public static void main(String[] args) {
        WikiPageCollectorService collector = new WikiPageCollectorService();
        String namespace = "de:modul:ffit:3-jahr:java:learningunits:lu11:";
        Set<String> pages = collector.collectPagesForNamespace("https://wiki.bzz.ch", namespace);
        log.info(pages.toString());
    }
}
